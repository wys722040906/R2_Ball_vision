#include "ros/ros.h"

int main(int argc, char **argv)
{
    // 初始化ROS节点
    ros::init(argc, argv, "parameter_modifier_node");

    // 创建节点句柄
    ros::NodeHandle nh;

    // 读取参数
    int parameter_value;
    nh.getParam("/your_node_name/parameter_name", parameter_value);

    // 修改参数
    parameter_value = 123;

    // 写入参数
    nh.setParam("/your_node_name/parameter_name", parameter_value);

    // 循环等待ROS事件
    ros::spin();

    return 0;
}
